        var Minimal_Angle = (h, m) => 
        {
          h=h%12;
          let min=(m*360)/60;
          let hour=(h*360)/12+(m*360)/(12*60);
          let angle=Math.abs(hour-min);
          
          if(angle>180){
            return 360-angle;
          }
          else{
            return angle;
          }
          
        };

Perfect number check

 
        var Perfect_Check = (N) => 
        {
        
          let sum=0;
        for(let i=0;i < N;i++){
          if(N%i===0){
            sum+=i;
          }
        }
        
        if(sum===N){
          return "YES";
        }
        else if(N===1){
          return "YES";
        }
        else{
        return "NO";
        }
        };

Reverse the number

 
        var Reverse_Number = (N) => 
        {
          let str=N.toString();
          let finalres="";
          for(let j=str.length-1;j>=0;j--){
            if(str[j]!==0){
              finalres+=str[j];
            }
            else if(str[j]===0){
              finalres+=null;
              
            }
          }
          return Math.floor(finalres);
        
        };
